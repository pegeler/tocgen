Make Tables of Contents

Generates a Table of Contents for Markdown and HTML documents. Input and output formats can be mixed and matched. The table can be output to STDOUT or written to file.

It is also extensible to any additional format you might want to add by subclassing the abstract base classes for parsing input files and generating output tables.

This is non-production software that is still under development. User input is not carefully validated. Suggested usage is in interactive workflows when editing your own documents. Use at your own risk.

 52    """
 53    Abstract base class for parsing document headings into a collection of
 54    entries. A subclass of the BaseTocGenerator can use those entries to
 55    generate a table of contents string.
 56    """
 57
 58    @abc.abstractmethod
 59    def parseFile(self, infile) -> list[TocEntry]:
 60        """
 61        Parse a file to create a list of TocEntries
 62
 63        :param str infile: The file to parse.
 64        """
 65        raise NotImplementedError
 66
 67    @classmethod
 68    def __subclasshook__(cls, c):
 69        if cls is BaseSimpleDocumentParser:
 70            if any('parseFile' in b.__dict__ for b in c.__mro__):
 71                return True

 75class SimpleHtmlParser(HTMLParser, BaseSimpleDocumentParser):
 76    """
 77    A parser that will generate a list of TOC entries from heading nodes,
 78    (*e. g.*, ``<h2>Foo</h2>``, ``<h3>Bar</h3>``) in an HTML document.
 79
 80    :ivar entries: A list of TOC entries from the most recently parsed file.
 81    """
 82    entries: list[TocEntry]
 83    _depth: int
 84    _heading: str
 85    _link: str
 86
 87    RE_HEADING = re.compile(r'^h(\d+)$', re.I)
 88
 89    def __init__(self, **kwargs):
 90        super().__init__(**kwargs)
 91        self._reset()
 92
 93    def _reset(self):
 94        self.entries = []
 95        self._resetTagVariables()
 96
 97    def _resetTagVariables(self):
 98        self._in_heading_tag = False
 99        self._heading = ''
100
101    def handle_starttag(self, tag, attrs):
102        if not self._in_heading_tag and (h := self.RE_HEADING.match(tag)):
103            self._in_heading_tag = True
104            self._depth = int(h.group(1)) - 1
105            for name, value in attrs:
106                if name == 'id':
107                    self._link = f'#{value}'
108                    break
109            else:
110                self._link = ''
111
112    def handle_endtag(self, tag):
113        if self._in_heading_tag and self.RE_HEADING.match(tag):
114            self.entries.append(TocEntry(self._depth, self._heading, self._link))
115            self._resetTagVariables()
116
117    def handle_data(self, data):
118        if self._in_heading_tag:
119            self._heading += data.replace('\n', '')
120
121    def parseFile(self, infile) -> list[TocEntry]:
122        self._reset()
123        with open(infile, 'rt') as fh:
124            self.feed(fh.read())
125        return self.entries

128class SimpleMarkdownParser(BaseSimpleDocumentParser):
129    """
130    A parser that will create a list of TOC entries from heading lines
131    (*e. g.*, ``## Foo``, ``### Bar``) in a Markdown document. Note that top
132    level headings (``# Main title``) will be ignored.
133    """
134    FENCE = '```'
135    RE_CAPTURE = re.compile('^(#+)(.*)$')
136    RE_HTML_COMMENT = re.compile(r'<!--.*?-->', flags=re.S)  # Non-greedy match
137    RE_SPECIALS = re.compile(r'''[!@#$%^&*()+;:'"\[\]{}|\\<>,./?`~]''')
138    RE_CUSTOM_ID = re.compile(r'''(.*?)\s*\{(#.+?)}''')
139
140    def __init__(self, *, use_custom_anchors=False):
141        self.use_custom_anchors = use_custom_anchors
142
143    def _openWithStrippedHtmlComments(self, filename: str) -> Iterator[str]:
144        with open(filename) as f:
145            yield from self.RE_HTML_COMMENT.sub('', f.read()).split('\n')
146
147    def _deriveLinkFromHeading(self, heading, links):
148        link = self.RE_SPECIALS.sub('', heading.lower().replace(' ', '-'))
149        links[link] += 1
150        if n_links := links[link] - 1:
151            link = f'{link}-{str(n_links)}'
152        return f'#{escape(link)}'
153
154    def parseFile(self, infile) -> list[TocEntry]:
155        in_fence = False
156        links: dict[str, int] = defaultdict(int)
157        entries = []
158
159        for line in self._openWithStrippedHtmlComments(infile):
160            if line.startswith(self.FENCE):
161                in_fence = not in_fence
162                continue
163
164            if in_fence or not (m := self.RE_CAPTURE.match(line)):
165                continue
166
167            depth = len(m.group(1)) - 2
168            heading = m.group(2).strip()
169            if depth < 0:
170                sys.stderr.write(f'Top level headings are ignored. '
171                                 f'Skipping {heading}\n')
172                continue
173
174            if (self.use_custom_anchors
175                    and (m := self.RE_CUSTOM_ID.match(heading))):
176                heading, link = m.groups()
177            else:
178                link = self._deriveLinkFromHeading(heading, links)
179
180            entries.append(TocEntry(depth, heading, link))
181
182        return entries
183
184
185def parse_file(infile: str, **kwargs) -> list[TocEntry]:
186    _, ext = os.path.splitext(infile.lower())
187
188    match ext:
189        case "md":
190            use_custom_anchors = kwargs.get("use_custom_anchors", False)
191            parser = SimpleMarkdownParser(use_custom_anchors=use_custom_anchors)
192        case "html" | "htm" | "xhtml":
193            parser = SimpleHtmlParser()
194        case _:
195            raise ValueError("Infile format not supported")
196
197    return parser.parseFile(infile)

200class BaseTocGenerator(abc.ABC):
201    """
202    Abstract base class for generating a Table of Contents. Must be subclassed
203    to generate a Table of Contents string for the corresponding output format.
204    """
205
206    def __init__(self, entries, indent=4, outfile=None):
207        self.entries = entries
208        self.indent = indent
209        self.indent_str = indent * ' '
210        self.outfile = outfile
211
212    def write(self):
213        """
214        Write the Table of Contents to file or STDOUT.
215        """
216        s = self.generateString()
217        if self.outfile:
218            with open(self.outfile, 'wt') as fh:
219                fh.write(s + '\n')
220        else:
221            print(s)
222
223    @abc.abstractmethod
224    def generateString(self) -> str:
225        """
226        Generate a Table of Contents string corresponding to the format.
227        """
228        raise NotImplementedError

231class MarkdownTocGenerator(BaseTocGenerator):
232
233    def generateString(self) -> str:
234        lines = ['## Table of Contents', '']
235        for entry in self.entries:
236            pad = entry.depth * self.indent_str
237            if entry.link:
238                heading_str = f'[{entry.heading}]({entry.link})'
239            else:
240                heading_str = entry.heading
241            lines.append(f'{pad}* {heading_str}')
242        return '\n'.join(lines)

245class HtmlTagGenerator(SimpleNamespace):
246    """
247    A variant of a SimpleNamespace that can be used to wrap strings in HTML
248    tags. For example,
249
250    >>> html_tags = HtmlTagGenerator(['h2', 'p', 'a'])
251    >>> html_tags.h2('This is a heading')
252    '<h2>This is a heading</h2>'
253    >>> mylink = html_tags.a("mylink", attrs=[("href", "#somewhere-in-doc")])
254    >>> html_tags.p(f'This is a link: {mylink}')
255    '<p>This is a link: <a href="#somewhere-in-doc">mylink</a></p>'
256
257    The instance is callable, resulting in a closure that can then be called
258    to wrap strings with the tag provided. Additional tags can also be defined
259    on the fly by using dot notation. Examples,
260
261    >>> html_tags('ol')(html_tags.li('one') + html_tags.li('two'))
262    '<ol><li>one</li><li>two</li></ol>'
263
264    """
265    def __init__(self, tags: Iterable[str]):
266        """
267        :param tags: The tags to be included. Accessed by dot notation.
268        """
269        super().__init__(**{t: self(t) for t in tags})
270
271    def __getattr__(self, item):
272        f = self(item)
273        self.__setattr__(item, f)
274        return f
275
276    def __call__(self, tag: str):
277
278        def html_tag(content: str,
279                     *,
280                     attrs: HtmlTagAttrs | None = None,
281                     newline: bool = False,
282                     indent: str | None = None):
283            """
284            :param content: The content to be placed inside the tag.
285            :param attrs: The attributes to be inserted into the tag.
286            :param newline: Whether to add a newline between tags and content.
287            :param indent: Pad left the tags with this string.
288            """
289            attr_str = ' ' + ' '.join(f'{k}="{v}"' for k, v in attrs) \
290                if attrs else ''
291            newline_str = '\n' if newline else ''
292            indent_str = indent or ''
293            return (indent_str + f'<{tag}{attr_str}>' + newline_str +
294                    content + newline_str +
295                    indent_str + f'</{tag}>')  # yapf: disable
296
297        return html_tag

300class HtmlTocGenerator(BaseTocGenerator):
301
302    def generateString(self) -> str:
303        return (self._wrapInTag.h2('Table of Contents') + '\n' +
304                self._generateUlStr(self.entries, 0))
305
306    def _generateUlStr(self, entries: Iterable[TocEntry], depth: int) -> str:
307        """
308        Generate a string containing potentially nested unordered list
309        (``<ul>``) nodes from a list of TOC entries.
310        """
311        output = []
312        groups = groupby(entries, lambda x: x.depth == depth)
313        for is_same_depth, entry_group in groups:
314            if is_same_depth:
315                for entry in entry_group:
316                    li = self._wrapInTag.li(self._maybeWrapInLink(entry))
317                    output.append((1 + depth) * self.indent_str + li)
318            else:
319                output.append(self._generateUlStr(entry_group, depth + 1))
320        return self._wrapInTag.ul(
321            '\n'.join(output), newline=True, indent=depth * self.indent_str)
322
323    def _maybeWrapInLink(self, entry: TocEntry) -> str:
324        """
325        Create a string out of a TOC entry. If the entry contains a non-empty
326        link attribute, it will be wrapped in an anchor tag (``<a>``).
327        """
328        if entry.link:
329            attrs = [('href', entry.link)]
330            return self._wrapInTag.a(entry.heading, attrs=attrs)
331        return entry.heading
332
333    _wrapInTag = HtmlTagGenerator(['h2', 'ul', 'li', 'a'])

Create a collection of name/value pairs.

Example enumeration:

>>> class Color(Enum):
...     RED = 1
...     BLUE = 2
...     GREEN = 3

Access them by:

  • attribute access::
>>> Color.RED
<Color.RED: 1>
  • value lookup:
>>> Color(1)
<Color.RED: 1>
  • name lookup:
>>> Color['RED']
<Color.RED: 1>

Enumerations can be iterated over, and know how many members they have:

>>> len(Color)
3

>>> list(Color)
[<Color.RED: 1>, <Color.BLUE: 2>, <Color.GREEN: 3>]

Methods can be added to enumerations, and members can have their own attributes -- see the documentation for details.

303    def generateString(self) -> str:
304        return (self._wrapInTag.h2('Table of Contents') + '\n' +
305                self._generateUlStr(self.entries, 0))
306
307    def _generateUlStr(self, entries: Iterable[TocEntry], depth: int) -> str:
308        """
309        Generate a string containing potentially nested unordered list
310        (``<ul>``) nodes from a list of TOC entries.
311        """
312        output = []
313        groups = groupby(entries, lambda x: x.depth == depth)
314        for is_same_depth, entry_group in groups:
315            if is_same_depth:
316                for entry in entry_group:
317                    li = self._wrapInTag.li(self._maybeWrapInLink(entry))
318                    output.append((1 + depth) * self.indent_str + li)
319            else:
320                output.append(self._generateUlStr(entry_group, depth + 1))
321        return self._wrapInTag.ul(
322            '\n'.join(output), newline=True, indent=depth * self.indent_str)
323
324    def _maybeWrapInLink(self, entry: TocEntry) -> str:
325        """
326        Create a string out of a TOC entry. If the entry contains a non-empty
327        link attribute, it will be wrapped in an anchor tag (``<a>``).
328        """
329        if entry.link:
330            attrs = [('href', entry.link)]
331            return self._wrapInTag.a(entry.heading, attrs=attrs)
332        return entry.heading
